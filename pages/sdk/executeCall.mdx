# prepareExecuteCall

Executes an encoded transaction from an ERC-6551 account.

<Tabs items={["viem", "Ethers.js"]}>
  <Tab>
    ```javascript
    import { executeCall } from "@tokenbound/sdk";

    const to = "0xe7134a029cd2fd55f678d6809e64d0b6a0caddcb"; // any address
    const value = 0n; // amount of ETH to send
    const data = ""; // calldata

    const hash = await executeCall(
      accountAddress,
      to,
      value,
      data,
      walletClient
    );
    ```
  </Tab>
  <Tab>
    ```javascript
    import { executeCall } from "@tokenbound/sdk-ethers";

    const to = "0xe7134a029cd2fd55f678d6809e64d0b6a0caddcb"; // any address
    const value = 0; // amount of ETH to send
    const data = "0x"; // calldata

    const hash = await executeCall(
      accountAddress,
      to,
      value,
      data,
      signer
    );
    ```
  </Tab>
</Tabs>
